\begin{frame}{Collapse creates an extreme environment}
  \begin{columns}[T,onlytextwidth]

    \begin{column}{0.58\textwidth}
      \centering
      % images/neutron_star_fieldlines.png is from https://www.cv.nrao.edu/~sransom/web/Ch6.html
      \vspace*{-0.15cm}
      % https://imagine.gsfc.nasa.gov/science/objects/neutron_stars1.html
      \includegraphics[width=\textwidth, keepaspectratio]{images/neutron_star_fieldlines_low_level.png}
    \end{column}

    \begin{column}{0.38\textwidth}
      \vspace{0.25cm}
      \begin{itemize}
        \item Collapse shrinks the radius by many orders of magnitude:
        \[
          R_{\star} \gg R_{\rm NS}
        \]

        \item Angular momentum conservation spins the remnant up:
        \[
          I\Omega \approx \text{const.}
          \quad\Rightarrow\quad
          \Omega_{\rm NS} \gg \Omega_{\star}
        \]

        \item Magnetic flux is approximately compressed:
        \[
          \Phi_B \sim B R^2 \approx \text{const.}
          \quad\Rightarrow\quad
          B_{\rm NS} \propto R^{-2}
        \]

        \item For Vela:
        \(f \simeq \qty{11}{\hertz}\), 
        \(B \sim \qty{e8}{\tesla}\)
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Why is a spectral cutoff expected?}

  \begin{columns}[T,onlytextwidth]

    \begin{column}{0.56\textwidth}

      \centering
      %https://pos.sissa.it/241/027/pdf
      \includegraphics[width=\textwidth]{images/cutoff.png}

      \vspace{0.15cm}

      {\scriptsize Example Plot to illustrate the expected gamma-ray spectrum and its cutoff}

    \end{column}

    \begin{column}{0.40\textwidth}

      \vspace{0.25cm}

      \begin{itemize}

        \item Pulsars efficiently accelerate charged particles

        \vspace{0.25cm}

        \item Curved relativistic motion produces strong radiation losses

        \vspace{0.25cm}

        \item Radiation losses increase rapidly at high energies

        \vspace{0.25cm}

        \item Standard curvature-radiation models therefore predict a cutoff around the GeV range

      \end{itemize}

      \vspace{0.45cm}

      \begin{block}{Key idea}
        The same processes that produce gamma rays also limit the maximum photon energy.
      \end{block}

    \end{column}

  \end{columns}

\end{frame}

\begin{frame}{The light cylinder}
  \begin{columns}[T,onlytextwidth]

    \begin{column}{0.58\textwidth}
      \centering
      % https://www.annualreviews.org/content/journals/10.1146/annurev-astro-052920-112338
      \vspace*{-0.35cm}
      \includegraphics[
        height=\textheight,
        keepaspectratio,
        trim={0 10cm 85cm 10cm},
        clip
      ]{images/neutron_star_fieldlines_extendet.png}
    \end{column}

    \begin{column}{0.38\textwidth}
      \vspace{0.05cm}

      \begin{block}{From co-rotation to a limit}
        If the magnetosphere co-rotates with the neutron star,
        the linear speed increases with distance from the rotation axis.
      \end{block}

      \vspace{0.15cm}

      \[
        R_{\mathrm{LC}} = \frac{c}{\omega}
      \]

      \vspace{0.2cm}

      \begin{itemize}
        \item At this radius, co-rotation would require the speed of light

        \vspace{0.18cm}

        \item Beyond it, rigid co-rotation is impossible

        \vspace{0.18cm}

        \item Magnetic field lines open up and particles can escape outward
      \end{itemize}

    \end{column}

  \end{columns}
\end{frame}

\begin{frame}{The High Energy Stereoscopic System (H.E.S.S.)}

  \begin{columns}[T,onlytextwidth]

    \begin{column}{0.58\textwidth}

      \centering
      %https://en.wikipedia.org/wiki/High_Energy_Stereoscopic_System
      \includegraphics[width=\textwidth]{images/Hess_array.jpg}

      \vspace{0.15cm}

      {\scriptsize The H.E.S.S. telescope array in Namibia}

    \end{column}

    \begin{column}{0.38\textwidth}
      \begin{itemize}

        \item Located in the Khomas Highland of Namibia

        \vspace{0.1cm}

        \item Array of Imaging Atmospheric Cherenkov Telescopes (IACTs)

        \vspace{0.1cm}

        \item Designed to observe very-high-energy gamma rays

        \vspace{0.1cm}

        \item Sensitive roughly from
        \qty{30}{\giga\electronvolt}
        to beyond
        \qty{10}{\tera\electronvolt}

        \vspace{0.1cm}

        \item Operates since 2003

      \end{itemize}

      \vspace{0.25cm}

      \begin{block}{Why H.E.S.S.?}
        Space-based gamma-ray telescopes become limited at very high energies.
        Ground-based Cherenkov telescopes allow us to study the TeV regime.
      \end{block}

    \end{column}

  \end{columns}

\end{frame}

\begin{frame}{How does H.E.S.S. detect TeV gamma rays?}

  \begin{columns}[T,onlytextwidth]

    \begin{column}{0.58\textwidth}

      \centering
      % https://skyandtelescope.org/sky-and-telescope-magazine/magic-in-the-air/
      \includegraphics[width=\textwidth]{images/Hess.png}

      \vspace{0.15cm}

      {\scriptsize Gamma rays create atmospheric particle showers that emit Cherenkov light}

    \end{column}

    \begin{column}{0.38\textwidth}

      \begin{itemize}

        \item TeV gamma rays do not reach the ground directly

        \vspace{0.25cm}

        \item They interact with the atmosphere and produce particle cascades

        \vspace{0.25cm}

        \item Fast charged particles emit short Cherenkov light flashes

        \vspace{0.25cm}

        \item H.E.S.S. reconstructs the gamma-ray direction, energy, and arrival time

      \end{itemize}

      \vspace{0.25cm}

      \begin{block}{Timing analysis}
        Using the pulsar rotation period, photon arrival times can be phase-folded to search for pulsed emission.
      \end{block}
    \end{column}

  \end{columns}

\end{frame}

\begin{frame}{Why is the H.E.S.S. Vela signal significant?}

  \begin{columns}[T,onlytextwidth]

    \begin{column}{0.60\textwidth}
      \centering
      \renewcommand{\arraystretch}{1.25}
      \normalsize
      \begin{tabular}{l c c c}
        \toprule
        \textbf{Energy cut} & \textbf{C-test} & \textbf{LR test} & \textbf{H-test} \\
        \midrule
        $>1$ TeV  & $4.3\sigma$ & -- & -- \\
        $>3$ TeV  & $4.9\sigma$ & $4.7\sigma$ & -- \\
        $>5$ TeV  & $\mathbf{5.8\sigma}$ & -- & -- \\
        $>7$ TeV  & $5.6\sigma$ & $4.8\sigma$ & $4.5\sigma$ \\
        $>20$ TeV & $4.6\sigma$ & -- & -- \\
        \bottomrule
      \end{tabular}

      \vspace{0.35cm}

      \begin{block}{Main point}
        Multiple independent periodicity tests give $>4\sigma$ evidence for pulsed emission.
      \end{block}

    \end{column}

    \begin{column}{0.37\textwidth}
      \small

      \textbf{Tests used}
      \vspace{0.1cm}

      \begin{itemize}
        \item H-test: no assumed pulse shape
        \item C-test: expected P2 pulse position/width
        \item LR test: ON/OFF phase regions
      \end{itemize}

      \vspace{0.25cm}

      \begin{block}{Why this matters}
      The TeV excess appears at the Vela P2 phase:
      \[
      \phi^{\rm TeV}_{P2}=0.568\pm0.003
      \]
      consistent with GeV P2:
      \[
      \phi^{\rm GeV}_{P2}=0.565\pm0.001
      \]
      \end{block}

    \end{column}

  \end{columns}

\end{frame}

\begin{frame}{How to read the tested Vela models}
  \small

  \begin{block}{Common assumption}
    Multi-TeV photons are produced by inverse-Compton scattering (IC) in all models:
    \(e^\pm+\gamma_{\rm seed}\rightarrow e^\pm+\gamma_{\rm TeV}\).\\
    The phase-aligned GeV and TeV components are treated as emission from one
    electron population with \(\gamma_{\max}\sim10^7\text{--}10^8\), where
    \(\gamma_{\max}\) is the maximum electron Lorentz factor.
  \end{block}

  \vspace{-0.1cm}

  \begin{columns}[T,onlytextwidth]

    \column{0.49\textwidth}
    \begin{block}{Class I: GeV = curvature radiation}
      \[
      E_{\rm CR}\propto \frac{\gamma^3}{\rho_c}
      \]
      {\scriptsize
      \(E_{\rm CR}\): characteristic curvature-radiation photon energy\\
      \(\rho_c\): curvature radius of the particle trajectory
      }

      \vspace{0.08cm}

      \begin{itemize}
        \item \(I_a\): reference CR model fitted to the GeV component only;
        the associated IC emission is then compared to the TeV data.
        
        \item \(I_b\): joint CR+IC description, where the same electron distribution
        is used for the GeV curvature-radiation and TeV IC components.
        
        \item \(I_c\): same CR+IC setup as \(I_b\), but with an extended seed-photon
        field including far-infrared photons.
      \end{itemize}
    \end{block}

    \column{0.49\textwidth}
    \begin{block}{Class II: GeV = synchrotron radiation}
      \[
      \text{current sheet / striped wind scenario}
      \]

      \vspace{-0.15cm}

      \begin{itemize}
        \item \(II_a\): reference SR model fitted to the GeV component only;
        the associated IC emission is then compared to the TeV data.
        
        \item \(II_b\): SR+IC model with a high-energy particle tail, motivated by
        escape from strong-cooling regions or reacceleration.
        
        \item \(II_c\): SR+IC model including relativistic bulk motion, so that the
        emitted spectrum is Doppler-boosted in the observer frame.
      \end{itemize}
    \end{block}

  \end{columns}

  \vspace{0.05cm}
  {\scriptsize
  CR: curvature radiation \quad
  SR: synchrotron radiation \quad
  IC: inverse-Compton scattering
  }

\end{frame}

      \begin{itemize}

        \item Particle acceleration near the current sheet

        \vspace{0.25cm}

        \item Magnetic reconnection in the outer magnetosphere

        \vspace{0.25cm}

        \item Inverse Compton scattering in the striped wind region

        \vspace{0.25cm}

        \item Emission produced farther from the neutron-star surface

      \end{itemize}
